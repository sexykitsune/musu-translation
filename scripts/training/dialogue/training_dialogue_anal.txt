
macro	training_dialogue_anal
select_icon_init
if training_chara=TRAINING_MIKAN	then gosub training_dialogue_anal_mikan
if training_chara=TRAINING_ICHIGO	then gosub training_dialogue_anal_ichigo
if training_chara=TRAINING_KARIN	then gosub training_dialogue_anal_karin
window_close
endm

label	training_dialogue_anal_mikan

skip_untranslated_training=0
if t_mikan_anal_cnt=0 then gosub training_dialogue_anal_mikan_01
if t_mikan_anal_cnt=1 then gosub training_dialogue_anal_mikan_02
if t_mikan_anal_cnt=2 then gosub training_dialogue_anal_mikan_03
if t_mikan_anal_cnt>2 then gosub training_dialogue_anal_mikan_04

if skip_untranslated_training=1 then return

return

label	training_dialogue_anal_ichigo

skip_untranslated_training=0
if t_ichigo_anal_cnt=0 then gosub training_dialogue_anal_ichigo1
if t_ichigo_anal_cnt=1 then gosub training_dialogue_anal_ichigo2
if t_ichigo_anal_cnt=2 then gosub training_dialogue_anal_ichigo3
if t_ichigo_anal_cnt>2 then gosub training_dialogue_anal_ichigo4

if skip_untranslated_training=1 then return

return

label	training_dialogue_anal_karin

skip_untranslated_training=0
if t_karin_anal_cnt=0 then gosub training_dialogue_anal_karin_01
if t_karin_anal_cnt=1 then gosub training_dialogue_anal_karin_02
if t_karin_anal_cnt=2 then gosub training_dialogue_anal_karin_03
if t_karin_anal_cnt>2 then gosub training_dialogue_anal_karin_04

if skip_untranslated_training=1 then return

return
