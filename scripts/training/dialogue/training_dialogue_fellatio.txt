
macro	training_dialogue_fellatio
skip_untranslated_training=0
select_icon_init
if training_chara=TRAINING_MIKAN	then gosub training_dialogue_fellatio_mikan
if training_chara=TRAINING_ICHIGO	then gosub training_dialogue_fellatio_ichigo
if training_chara=TRAINING_KARIN	then gosub training_dialogue_fellatio_karin
window_close
if skip_untranslated_training=1 then return
@training_anime_load
endm

label	training_dialogue_fellatio_mikan
sel=0
if _COLOR=cl_mouth then sel=0
if _COLOR=cl_face  then sel=1

if training_fellatio_count=0 then gosub training_dialogue_fellatio_mikan_01
if training_fellatio_count=1 then gosub training_dialogue_fellatio_mikan_02
if training_fellatio_count=2 then gosub training_dialogue_fellatio_mikan_03
if training_fellatio_count>2 then gosub training_dialogue_fellatio_mikan_04
return

macro	t_mikan_fera_sub
if sel=1 then goto training_dialogue_fellatio_mikan_mouth
if sel=0 then goto training_dialogue_fellatio_mikan_face

label	training_dialogue_fellatio_mikan_mouth
if training_fellatio_mouth_count=0 then gosub training_dialogue_fellatio_mikan_a01
if training_fellatio_mouth_count=1 then gosub training_dialogue_fellatio_mikan_a02
if training_fellatio_mouth_count=2 then gosub training_dialogue_fellatio_mikan_a03
if training_fellatio_mouth_count>2 then gosub training_dialogue_fellatio_mikan_a04
training_fellatio_mouth_count=training_fellatio_mouth_count+1
goto	training_dialogue_fellatio_mikan_skip

label	training_dialogue_fellatio_mikan_face
if training_fellatio_face_count=0 then gosub training_dialogue_fellatio_mikan_b01
if training_fellatio_face_count=1 then gosub training_dialogue_fellatio_mikan_b02
if training_fellatio_face_count=2 then gosub training_dialogue_fellatio_mikan_b03
if training_fellatio_face_count>2 then gosub training_dialogue_fellatio_mikan_b04
training_fellatio_face_count=training_fellatio_face_count+1

label	training_dialogue_fellatio_mikan_skip
endm

label	training_dialogue_fellatio_ichigo
if training_fellatio_count=0 then gosub training_dialogue_fellatio_ichigo_01
if training_fellatio_count=1 then gosub training_dialogue_fellatio_ichigo_02
if training_fellatio_count=2 then gosub training_dialogue_fellatio_ichigo_03
if training_fellatio_count>2 then gosub training_dialogue_fellatio_ichigo_04
return

label	training_dialogue_fellatio_karin
if training_fellatio_count=0 then gosub training_dialogue_bukkake_karin
if training_fellatio_count=1 then gosub training_dialogue_hairjob_karin
if training_fellatio_count=2 then gosub training_dialogue_nosehook_karin
if training_fellatio_count=3 then gosub training_dialogue_fellatio_karin_01
if training_fellatio_count=4 then gosub training_dialogue_fellatio_karin_02
if training_fellatio_count=5 then gosub training_dialogue_fellatio_karin_03
if training_fellatio_count=6 then gosub training_dialogue_fellatio_karin_04
if training_fellatio_count>6 then gosub training_dialogue_fellatio_karin_05
return
